1 #!/usr/bin/env ruby
 2 require 'rubygems'
 3 require 'nokogiri'
 4 require 'open-uri'
 5 require 'thor'
 6 
 7 class PictureDocument
 8 
 9   def initialize url, domain_name
10     @document = Nokogiri::HTML(open(url))
11     @domain_name = domain_name
12   end
13 
14   def image_url
15     download_link = @document.css('.download_link a')
16     if download_link.empty?
17       return nil
18     else
19       return download_link.first.attributes["href"].value
20     end
21   end
22 
23   def next_page_url
24     next_page_url = @domain_name + @document.css('.prev a').first.attributes["href"].value
25     puts next_page_url
26     return next_page_url
27   end
28 end
29 
30 class NationalGeographicDownloader < Thor
31 
32   LAST_DOWNLOADED_PATH = '.last_download'
33 
34 
35   desc 'download', 'Downloads pictures'
36   method_option :number_of_pictures, :type => :numeric, :aliases => '-n', :default => 10
37   method_option :starting_url, :type => :string, :aliases => '-s', :default => "http://photography.nationalgeographic.com/photography/photo-of-the-day"
38   method_option :resume_from_last_download, :type => :boolean, :aliases => '-r', :default => false
39   method_option :download_dir, :type => :string, :aliases => '-d', :default => './'
40   method_options :force_overwrite => false
41   def download
42     domain_name = "http://photography.nationalgeographic.com"
43     if options.resume_from_last_download
44       begin
45         File.open(LAST_DOWNLOADED_PATH, 'r') do |file|
46           @url = file.gets
47           puts @url
48         end
49       rescue Exception => e
50         puts "Cannot retrieve last downloaded url, are you sure you already ran this script at least once?"
51         exit 1
52       end
53     else
54       @url = options.starting_url
55     end
56     number_of_downloaded_pictures = 0
57     while number_of_downloaded_pictures <= options.number_of_pictures do
58       picture_document = PictureDocument.new(@url, domain_name)
59       image_url = picture_document.image_url
60       if image_url
61         # download picture
62         print "Download link found in page at url #{@url}, proceeding..."
63         target_path = File.join(options.download_dir, image_url.split('/').last)
64         if File.exist?(target_path) && (options.force_overwrite != true)
65           puts "Image #{target_path} already downloaded, exiting! Bye!"
66           exit
67         end
68         system("wget #{image_url} -O #{target_path}  2>/dev/null")
69         puts " [DONE]"
70         number_of_downloaded_pictures = number_of_downloaded_pictures + 1
71       else
72         # skip and go to next page
73         puts "Page at url #{@url} has no download link, skipping..."
74       end
75       @url = picture_document.next_page_url
76     end
77     puts "Last downloaded picture was at url #{@url}"
78     File.open(LAST_DOWNLOADED_PATH, 'w') do |file|
79       file.puts @url
80     end
81   end
82 end
83 
84 NationalGeographicDownloader.start
